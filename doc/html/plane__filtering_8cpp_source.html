87 void KinectRawDepthCam::initDepthReconstructionLookups()
88 {
89  float h = 1.0/f;
90  float v = 1.0/f;
91  double tanHFovH = tan(fovH*0.5);
92  double tanHFovV = tan(fovV*0.5);
93  float x,y,z;
94  int ind;
95 
96  pointCloudLookups.resize(height*width);
97 
98  for(unsigned int row = 0; row<height; row++){
99  for(unsigned int col = 0; col<width; col++){
100  ind = row*width + col;
101  y = -(float(col)/((float(width)-1.0)*0.5)-1.0)*tanHFovH;
102  z = -(float(row)/((float(height)-1.0)*0.5)-1.0)*tanHFovV;
103  pointCloudLookups[ind] = vector3f(1.0f,y,z,1.0f);
104  }
105  }
106 
107  for(int depth = 0; depth<65536; depth++){
108  depthLookups[depth] = 1.0/(a+b*float(depth));
109  }
110 }

239 inline bool PlaneFilter::sampleLocation(void* depthImage, int& index, int& row, int& col, Vector3f& p, int rMin, int height, int cMin, int width)
240 {
241  unsigned int retries = 0;
242  bool valid = false;
243  while( !valid && retries<=filterParams.numRetries){
244  row = rMin + (lcgRand()%height);
245  col = cMin + (lcgRand()%width);
246  index = row*depthCam->width + col;
247  valid = depthCam->isValidDepth(index,depthImage);
248  numSampledLocations++;
249  retries++;
250  }
251  if(valid) p = Vector3f(V3COMP(depthCam->depthPixelTo3D(index,depthImage)));
252  return valid;
253 }

300 void PlaneFilter::GenerateFilteredPointCloud(void* depthImage, vector< vector3f >& filteredPointCloud, vector< vector2i >& pixelLocs, vector< vector3f >& pointCloudNormals, vector< vector3f >& outlierCloud, vector< PlanePolygon >& polygons)
301 {
302  static const bool debug = false;
303 
304  planeFilteringTimer.start();
305 
306  polygons.clear();
307  filteredPointCloud.clear();
308  pixelLocs.clear();
309  pointCloudNormals.clear();
310  outlierCloud.clear();
311 
312  filteredPointCloud.reserve(2*filterParams.maxPoints);
313  pixelLocs.reserve(2*filterParams.maxPoints);
314  pointCloudNormals.reserve(2*filterParams.maxPoints);
315  outlierCloud.reserve(filterParams.numSamples);
316 
317  // Derived parameters
318  float minInliers = filterParams.minInlierFraction*filterParams.numLocalSamples;
319  float maxOutliers = (1.0-filterParams.minInlierFraction)*filterParams.numLocalSamples;
320  float planeSizeH = 0.5*filterParams.planeSize;
321  float planeSize = filterParams.planeSize;
322  float w2 = depthCam->width-filterParams.planeSize;
323  float h2 = depthCam->height-filterParams.planeSize;
324  // Counters
325  unsigned int numPlanes = 0;
326  unsigned int numPoints = 0;
327 
328  // Sample locations
329  int ind1,ind2,ind3;
330 
331  float sampleRadiusHFactor = filterParams.WorldPlaneSize*depthCam->width/(4.0*tan(depthCam->fovH));
332  float sampleRadiusVFactor = filterParams.WorldPlaneSize*depthCam->height/(4.0*tan(depthCam->fovV));
333 
334  Vector3f p1, p2, p3, p;
335  vector2i l1, l2, l3, l;
336 
337  vector<vector3f> neighborhoodInliers;
338  vector<vector2i> neighborhoodPixelLocs;
339 
340  float d, meanDepth;
341  int sampleRadiusH, sampleRadiusV, rMin, rMax, cMin, cMax, dR, dC;
342 
343  for(unsigned int i=0; i<filterParams.numSamples && numPoints<filterParams.maxPoints; i++){
344 
345  //generate random point p1 anywhere on image
346  if(!sampleLocation(depthImage, ind1, l1.y, l1.x, p1, planeSizeH, h2, planeSizeH, w2))
347  continue;
348 
349  //generate random point p2 within max distance params.planeSize from p1
350  if(!sampleLocation(depthImage, ind2, l2.y, l2.x, p2, l1.y-planeSizeH, planeSize, l1.x-planeSizeH, planeSize))
351  continue;
352 
353  //generate random point p3 within max distance params.planeSize from p1
354  if(!sampleLocation(depthImage, ind3, l3.y, l3.x, p3, l1.y-planeSizeH, planeSize, l1.x-planeSizeH, planeSize))
355  continue;
356 
357  //Generate Plane normal (n) and distance (d) from origin (distance-normal parameterization of plane)
358  Vector3f n = ((p1-p2).cross(p3-p2)).normalized();
359 
360  d = p1.dot(n);
361  meanDepth = (p1.x()+p2.x()+p3.x())*0.333333333333333333333333;
362  sampleRadiusH = ceil(sampleRadiusHFactor/meanDepth*sqrt(1.0-sq(n.y())));
363  sampleRadiusV = ceil(sampleRadiusVFactor/meanDepth*sqrt(1.0-sq(n.z())));
364  rMin = max(0,l1.y-sampleRadiusV);
365  rMax = min(depthCam->height-1,l1.y+sampleRadiusV);
366  cMin = max(0,l1.x-sampleRadiusH);
367  cMax = min(depthCam->width-1,l1.x+sampleRadiusH);
368  dR = rMax-rMin;
369  dC = cMax-cMin;
370 
371  if(sampleRadiusH<2.0 || sampleRadiusV<2.0){
372  //printf("FSPF error %d,%d %f,%f,%f %f\u00b0, %f, %d, %f\n",sampleRadiusH,sampleRadiusV,n.x(),n.y(),n.z(),DEG(depthCam->fovH),filterParams.WorldPlaneSize,depthCam->width, meanDepth);
373  continue;
374  }
375 
376  int inliers = 0, outliers = 0;
377  neighborhoodInliers.clear();
378  neighborhoodPixelLocs.clear();
379  for(unsigned int j=0; outliers<maxOutliers && j<filterParams.numLocalSamples; j++){
380  //generate random point p within max distance params.planeSize from p1
381  int r,c,ind;
382  if(!sampleLocation(depthImage, ind, l.y, l.x, p, rMin, dR, cMin, dC))
383  continue;
384 
385  float err = fabs(n.dot(p) - d);
386  if(err<filterParams.maxError && p.x()<meanDepth+filterParams.maxDepthDiff && p.x()>meanDepth-filterParams.maxDepthDiff){
387  inliers++;
388  neighborhoodInliers.push_back(EIG2GV3(p));
389  neighborhoodPixelLocs.push_back(l);
390  }else{
391  outliers++;
392  }
393  }
394  if(inliers>=minInliers && inliers>3){
395  //==Polygonization==
396  if(filterParams.runPolygonization){
397  PlanePolygon poly(neighborhoodInliers,neighborhoodPixelLocs);
398  if(poly.validPolygon)
399  polygons.push_back(poly);
400  n = Vector3f(V3COMP(poly.normal));
401  }
402  //This is a local plane
403  vector3f ng = EIG2GV3(n);
404  ng.w = 0;
405  for(int i=0; i<(int)neighborhoodInliers.size(); i++){
406  //filteredPointCloud.push_back(pointCloud[i]);
407  filteredPointCloud.push_back(neighborhoodInliers[i]);
408  pointCloudNormals.push_back(ng);
409  pixelLocs.push_back(neighborhoodPixelLocs[i]);
410  }
411  filteredPointCloud.push_back(EIG2GV3(p1));
412  filteredPointCloud.push_back(EIG2GV3(p2));
413  filteredPointCloud.push_back(EIG2GV3(p3));
414  pointCloudNormals.push_back(ng);
415  pointCloudNormals.push_back(ng);
416  pointCloudNormals.push_back(ng);
417  pixelLocs.push_back(l1);
418  pixelLocs.push_back(l2);
419  pixelLocs.push_back(l3);
420  numPoints += neighborhoodInliers.size()+3;
421  numPlanes = numPlanes+1;
422 
423  }else{
424  for(int i=0; i<(int)neighborhoodInliers.size(); i++){
425  outlierCloud.push_back(neighborhoodInliers[i]);
426  }
427  outlierCloud.push_back(EIG2GV3(p1));
428  outlierCloud.push_back(EIG2GV3(p2));
429  outlierCloud.push_back(EIG2GV3(p3));
430  }
431  }
432  if(filterParams.runPolygonization && filterParams.filterOutliers){
433  //Remove planar points from outlier cloud
434  static const float MaxPlanarDist = filterParams.maxError;
435  bool planar = false;
436  for(unsigned int i=0; i<outlierCloud.size(); i++){
437  planar = false;
438  for(unsigned int j=0; !planar && j<polygons.size(); j++){
439  planar = fabs(polygons[j].normal.dot(outlierCloud[i])+polygons[j].offset)<MaxPlanarDist;
440  }
441  if(planar){
442  //Need to remove this point
443  outlierCloud.erase(outlierCloud.begin()+i);
444  i--;
445  }
446  }
447  }
448  numPlanarPoints = filteredPointCloud.size();
449  numNonPlanarPoints = outlierCloud.size();
450  planeFilteringTimer.stop();
451 }
