349 void VectorLocalization2D::computeParticleWeights(vector2f deltaLoc, float deltaAngle, vector2f minLocStdDev, float minAngleStdDev, const MotionModelParams &motionParams)
350 {
351  static const bool debug = false;
352 
353  int N = particles.size();
354  vector<vector2f> locMean(2*N);
355  vector<vector2f> locStdDev(2*N);
356  vector<float> angleMean(2*N);
357  vector<float> angleStdDev(2*N);
358 
359  currentLocStdDev.x = max(minLocStdDev.x,currentLocStdDev.x);
360  currentLocStdDev.y = max(minLocStdDev.y,currentLocStdDev.y);
361  currentAngleStdDev = max(minAngleStdDev,currentAngleStdDev);
362 
363  // Compute kernel supports for motion model weights
364  float deltaTranslation = deltaLoc.length();
365 
366  for(int i=0; i<N; i++){
367  angleMean[i] = particles[i].angle;
368  angleStdDev[i] = 1.0/sq( max( float(fabs(deltaAngle))*motionParams.Alpha1 + deltaTranslation*motionParams.Alpha2 , currentAngleStdDev) );
369  locMean[i] = particles[i].loc;
370  locStdDev[i].x = 1.0/sq(max( (deltaLoc.rotate(particles[i].angle).x)*motionParams.Alpha3, currentLocStdDev.x));
371  locStdDev[i].y = 1.0/sq(max( (deltaLoc.rotate(particles[i].angle).y)*motionParams.Alpha3, currentLocStdDev.y));
372  }
373  // Compute motion model weights from kernel supports
374  for(int i=0; i<N; i++){
375  Particle2D &p1 = particles[i];
376  Particle2D &p2 = particlesRefined[i];
377  p1.weight = 0.0;
378  p2.weight = 0.0;
379  for(int j=0; j<N; j++){
380  p1.weight += exp(-sq(p1.loc.x - locMean[j].x)*locStdDev[j].x -sq(p1.loc.y - locMean[j].y)*locStdDev[j].y -sq(angle_diff(p1.angle , angleMean[j]))*angleStdDev[j] );
381  p2.weight += exp(-sq(p2.loc.x - locMean[j].x)*locStdDev[j].x -sq(p2.loc.y - locMean[j].y)*locStdDev[j].y -sq(angle_diff(p2.angle , angleMean[j]))*angleStdDev[j] );
382  }
383  }
384 
385  if(debug){
386  printf("Motion model weights:\n");
387  for(int i=0; i<N; i++){
388  printf("%2d unrefined:%20.16f, %20.16f refined:%20.16f, %20.16f\n",i, particles[i].weight,stage0Weights[i], particlesRefined[i].weight,stageRWeights[i]);
389  }
390  }
391 
392  // Apply observation function weights
393  for(int i=0; i<N; i++){
394  particles[i].weight *= stage0Weights[i];
395  particlesRefined[i].weight *= stageRWeights[i];
396  }
397 
398  // Compute kernel supports for sampling density function
399  for(int i=0; i<N; i++){
400  angleMean[i] = particles[i].angle;
401  angleStdDev[i] = 1.0/sq( max( float(fabs(deltaAngle))*motionParams.Alpha1 + deltaTranslation*motionParams.Alpha2 , currentAngleStdDev) );
402  locMean[i] = particles[i].loc;
403  locStdDev[i].x = 1.0/sq(max( (deltaLoc.rotate(particles[i].angle).x)*motionParams.Alpha3, currentLocStdDev.x));
404  locStdDev[i].y = 1.0/sq(max( (deltaLoc.rotate(particles[i].angle).y)*motionParams.Alpha3, currentLocStdDev.y));
405 
406  angleMean[N+i] = particlesRefined[i].angle;
407  angleStdDev[N+i] = 1.0/sq( max( float(fabs(deltaAngle))*motionParams.Alpha1 + deltaTranslation*motionParams.Alpha2 , currentAngleStdDev) );
408  locMean[N+i] = particlesRefined[i].loc;
409  locStdDev[N+i].x = 1.0/sq(max( (deltaLoc.rotate(particlesRefined[i].angle).x)*motionParams.Alpha3, currentLocStdDev.x));
410  locStdDev[N+i].y = 1.0/sq(max( (deltaLoc.rotate(particlesRefined[i].angle).y)*motionParams.Alpha3, currentLocStdDev.y));
411  }
412  // Apply sampling density function weights from kernel supports
413  for(int i=0; i<N; i++){
414  Particle2D &p1 = particles[i];
415  Particle2D &p2 = particlesRefined[i];
416  float w1 = 0.0;
417  float w2 = 0.0;
418  for(int j=0; j<2*N; j++){
419  w1 += exp(-sq(p1.loc.x - locMean[j].x)*locStdDev[j].x -sq(p1.loc.y - locMean[j].y)*locStdDev[j].y -sq(angle_diff(p1.angle , angleMean[j]))*angleStdDev[j] );

421  }
422  p1.weight = p1.weight/w1;
423  p2.weight = p2.weight/w2;
424  }
425 
426  if(debug){
427  printf("Final weights:\n");
428  for(int i=0; i<N; i++){
429  printf("%2d unrefined:%20.16f refined:%20.16f\n",i, particles[i].weight, particlesRefined[i].weight);
430  }
